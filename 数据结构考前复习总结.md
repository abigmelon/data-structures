### 1. 万能模板 — 带头节点的单链表

```c
struct ListNode {
    int element;          // 数据域（可能是别的类型）
    struct ListNode *next;
};
typedef struct ListNode * Node;

// ① 初始化哨兵头节点
Node head = malloc(sizeof(struct ListNode));
head->next = NULL;

// ② 尾插法（保持顺序）
Node tail = head;
for (int i = 0; i < n; i++) {
    Node node = malloc(sizeof(struct ListNode));
    node->element = val;
    node->next = NULL;
    tail->next = node;
    tail = node;
}

// ③ 遍历
Node p = head->next;
while (p) {
    // 处理 p
    p = p->next;
}

// ④ 删除 p 的后继节点
Node tmp = p->next;
p->next = tmp->next;
free(tmp);
```

### 2. 链表排序（冒泡）

来自 **24-sliver.c**：

```c
void upSortList(Node head) {
    Node end = NULL;
    while (head->next != end) {       // 外层：逐步缩小范围
        Node tmp = head->next;
        while (tmp->next != end) {    // 内层：两两比较
            if (tmp->weight > tmp->next->weight) {
                // 交换 weight（或整个节点）
                int t = tmp->weight;
                tmp->weight = tmp->next->weight;
                tmp->next->weight = t;
            }
            tmp = tmp->next;
        }
        end = tmp;
    }
}
```

## 🔄 二、循环链表（必考！）

### 题一：`17-circle.c` — 循环删最小值

**题意**：输入 N 个数建成循环链表，每轮从当前节点遍历一圈找最小值，打印当前到最小值之间的值，删除最小值，从被删节点的下一个继续。直到只剩一个。

**完整流程**：

```c
// ===== 第一步：构建普通链表 =====
Node head = malloc(sizeof(struct listNode));
head->next = NULL;
Node tail = head;
for (int i = 0; i < n; i++) {
    Node node = malloc(sizeof(struct listNode));
    node->element = val;
    node->next = NULL;
    tail->next = node;
    tail = node;
}

// ===== 第二步：变成循环链表 =====
if (n == 0) { free(head); return 0; }
tail->next = head->next;       // 尾连头，形成环
Node current = head->next;     // 从第一个数据节点开始

// ===== 第三步：N 轮循环 =====
for (int round = 0; round < n; round++) {
    // 只剩一个节点
    if (current->next == current) {
        printf("%d \n", current->element);
        free(current);
        break;
    }

    // 找 current 的前驱
    Node prev_current = current;
    while (prev_current->next != current)
        prev_current = prev_current->next;

    // 遍历一圈找最小值及其前驱
    int min_val = current->element;
    Node min_node = current;
    Node min_prev = prev_current;
    Node p = current->next;
    Node p_prev = current;
    do {
        if (p->element < min_val) {
            min_val = p->element;
            min_node = p;
            min_prev = p_prev;
        }
        p_prev = p;
        p = p->next;
    } while (p != current);

    // 打印从 current 到 min_node
    p = current;
    do {
        printf("%d ", p->element);
        if (p == min_node) break;
        p = p->next;
    } while (1);
    printf("\n");

    // 删除 min_node，更新 current
    min_prev->next = min_node->next;
    current = min_node->next;
    free(min_node);
}
free(head);
```

### 题二：`23-spare.c` — 循环首次适应内存分配

**题意**：N 个空闲内存块连成循环链表，读入需求大小，从 current 开始循环查找第一个够大的块。恰好相等则删除，大于则缩小。输入 -1 结束。

```c
// ===== 构建循环链表 =====
Node first = head.next;
Node tail = findTailList(&head);
tail->next = first;            // 尾连头
Node current = first;

// ===== 循环处理需求 =====
int require;
while (scanf("%d", &require) != EOF) {
    if (require == -1) break;

    // 循环查找第一个满足条件的块
    Node found = NULL;
    Node p = current;
    if (p != NULL) {
        do {
            if (p->size >= require) {
                found = p;
                break;
            }
            p = p->next;
        } while (p != current);
    }

    if (found == NULL) continue;   // 找不到，忽略

    if (found->size == require) {
        // 恰好相等 → 删除节点
        Node tmp = current;
        while (tmp->next != found)
            tmp = tmp->next;
        current = (found == found->next) ? NULL : found->next;
        tmp->next = found->next;
        free(found);
    } else {
        // 大于需求 → 缩小
        found->size -= require;
        current = found;           // 下次从这开始！
    }
}
```

## 🔁 三、状态机模式（最高频技巧）

### 核心思想

用 `0/1` 标志位，控制"是否正在收集字符"。

```
遍历字符串，每个字符判断：
├── 遇到「开始条件」→ 标志位置 1，开始收集
├── 遇到「结束条件」→ 标志位置 0，保存结果
└── 标志位为 1     → 把当前字符加入缓冲区
```

### 通用模板

```c
int flag = 0;    // 标志位
int j = 0;       // 缓冲区写入位置

for (int i = 0; i < len; i++) {
    if (/* 开始条件 */) {
        flag = 1;
        buf[j++] = input[i];
    }
    else if (/* 结束条件 */) {
        flag = 0;
        buf[j] = '\0';       // 封口
        j = 0;
        // 处理 buf...
    }
    else if (flag == 1) {
        buf[j++] = input[i]; // 收集
    }
}
// ⚠️ 最后收尾！
if (flag == 1) {
    buf[j] = '\0';
    // 处理 buf...
}
```

### 题一：`20-mark.c` — 提取标识符并去重排序

```c
int copy = 0;
for (int i = 0; i < len; i++) {
    if (isalpha(input[i]) || input[i] == '_') {
        // 字母或下划线
        if (copy == 0) { copy = 1; }  // 开始新标识符
        output[cnt][index++] = input[i];
    }
    else if (isdigit(input[i])) {
        // 数字：只有在 copy=1 时才收集
        if (copy == 0) continue;
        output[cnt][index++] = input[i];
    }
    else {
        // 分隔符：结束当前标识符
        if (copy == 1) {
            output[cnt][index] = '\0';
            cnt++;
            index = 0;
        }
        copy = 0;
    }
}
if (copy == 1) {  // ⚠️ 收尾！
    output[cnt][index] = '\0';
    cnt++;
}
```

### 题二：`23-suoxie.c` — 连续递增字符缩写

```c
int start = 0;
for (int i = 0; i < len; i++) {
    // 开始：连续三个递增
    if (i+2 < len && input[i+1]-input[i]==1 &&
        input[i+2]-input[i+1]==1 && start == 0) {
        start = 1;
        output[j++] = input[i];
        continue;
    }
    // 中间：还在递增 → 跳过
    if (i+1 < len && input[i+1]-input[i]==1 && start == 1)
        continue;
    // 结束：递增断裂
    if (i+1 < len && input[i+1]-input[i]!=1 && start == 1) {
        start = 0;
        output[j++] = '-';
        output[j++] = input[i];
        continue;
    }
    // 普通字符
    if (start == 0) output[j++] = input[i];
}
if (start == 1) {  // ⚠️ 收尾！
    output[j++] = '-';
    output[j++] = input[len - 1];
}
```

### 题三：`24-hvref.c` — 括号内字符串去重

```c
int copy = 0, index = -1, j = 0;
for (int i = 0; i < len; i++) {
    if (input[i] == ',') continue;
    if (input[i] == '(') { copy = 1; index++; continue; }  // 开始
    if (input[i] == ')') { copy = 0; text[index][j]='\0'; j=0; continue; } // 结束
    if (copy == 1) text[index][j++] = input[i];             // 收集
}
```

### 三道状态机题对比

| | 20-mark | 23-suoxie | 24-hvref |
|------|:--:|:--:|:--:|
| 标志名 | `copy` | `start` | `copy` |
| 开始条件 | `isalpha` 或 `_` | 连续3个递增 | `(` |
| 结束条件 | 非字母数字下划线 | 递增断裂 | `)` |
| 收集中间内容 | ✅ 字母数字 | ❌ 跳过 | ✅ 全部 |
| 收尾处理 | ✅ 封口 `\0` | ✅ 补末字符 | ✅ 封口 `\0` |
| 后续处理 | 去重+排序 | 无 | 去重 |

## ⚡ 六、字符串函数详解（真题核心！）

### 6.0 C 语言字符串的本质

**字符串就是字符数组，以 `'\0'` 结尾**。所有 string.h 的函数都依赖 `'\0'` 来判断字符串在哪结束。如果你忘记写 `'\0'`，`strlen` 会一直读到内存里碰巧遇到 `'\0'` 为止，结果不可预测。

```c
char s1[] = "hello";    // 实际存储：h e l l o \0  → strlen = 5
char s2[3] = "hello";   // ❌ 数组太小，没有空间放 \0！
char s3[10] = "hi";     // 实际存储：h i \0 ? ? ? ? ? ? ?  → strlen = 2
char s4[10]; s4[0]='a'; // ❌ 没加 \0，strlen(s4) 不确定！
```

---

### 6.1 `strlen` — 求字符串长度

```c
#include <string.h>
size_t strlen(const char *s);
```

**作用**：返回从 `s` 开始到第一个 `'\0'` 之前的字符个数。**不包含 `'\0'` 本身**。

```c
char s[] = "abc";
strlen(s)  →  3     // 'a' 'b' 'c' 三个字符，\0 不计
strlen("")  →  0     // 空字符串

char t[100] = "hi";
strlen(t)  →  2     // 只算到 \0，不管数组有多大

// ⚠️ 常见陷阱
char u[5] = {'a', 'b', 'c', 'd', 'e'};  // 没有 \0！
strlen(u)  →  ???   // 读取越界，值不确定，程序可能崩溃！
```

**真题怎么用**：几乎每道字符串题开头都会 `int len = strlen(input);` 来确定遍历范围。

---

### 6.2 `strcmp` — 比较两个字符串 ⭐

```c
#include <string.h>
int strcmp(const char *s1, const char *s2);
```

**返回值**（这是最容易被问到的！）：

| 返回值 | 含义 |
|:------:|------|
| **0** | s1 和 s2 **完全相等** |
| **< 0** | s1 **字典序小于** s2 |
| **> 0** | s1 **字典序大于** s2 |

```c
strcmp("abc", "abc")   →  0      // 相等
strcmp("abc", "abd")   →  -1     // c < d，所以 abc < abd
strcmp("abc", "ab")    →  99     // 前两个相同，abc 更长 → 更大（返回 'c' 的ASCII）
strcmp("ABC", "abc")   →  -32    // 'A'=65, 'a'=97，大写 < 小写
strcmp("abc", "ABC")   →  32     // 反过来就是正数
```

**比较规则**：从左到右逐字符比较 ASCII 码，第一个不同字符决定大小；完全相同则返回 0。

**真题怎么用**：
```c
// 判断相等（去重）
if (strcmp(output[i], output[j]) == 0) { /* 重复了 */ }

// 排序（冒泡中判断是否需要交换）
if (strcmp(ans[j], ans[j + 1]) > 0) { /* 交换 */ }
```

> ⚠️ **不能用 `==` 比较字符串**！`s1 == s2` 比较的是内存地址，不是内容！

### 6.3 `strcpy` — 复制字符串

```c
#include <string.h>
char *strcpy(char *dest, const char *src);
```

**作用**：把 `src` 的内容（包括 `'\0'`）复制到 `dest`。返回 `dest` 的地址（一般不用）。

**真题怎么用**：


### 6.4 `strcat` — 拼接字符串

**作用**：把 `src` 追加到 `dest` 末尾（覆盖 `dest` 原来的 `'\0'`，末尾再加 `'\0'`）。

### 6.5 `strncpy` / `strncmp` / `strncat` — 安全版本

带 `n` 的版本都有一个额外参数限制操作长度，防止溢出：

```c
char *strncpy(char *dest, const char *src, size_t n);
int   strncmp(const char *s1, const char *s2, size_t n);
char *strncat(char *dest, const char *src, size_t n);
```

```c
// strncpy：最多复制 n 个字符
char a[10];
strncpy(a, "hello world", 9);  // 只复制前 9 个
a[9] = '\0';                   // ⚠️ 如果 src 比 n 长，strncpy 不会加 \0！

// strncmp：只比较前 n 个字符
strncmp("hello", "help", 3);   → 0    // 前 3 字符 "hel" 相同
strncmp("hello", "help", 4);   → < 0  // 第 4 字符 'l' < 'p'
```

### 6.9 `fgets` — 安全读一行 ⭐

```c
#include <stdio.h>
char *fgets(char *buf, int size, FILE *stream);
```

**作用**：从文件/标准输入读取最多 `size-1` 个字符，遇到换行符或 EOF 停止。

```c
char buf[100];
fgets(buf, 100, stdin);    // 从键盘读一行

// ⚠️ 核心问题：fgets 会保留换行符 '\n'！
// 真题中必须手动去掉：
int len = strlen(buf);
if (len > 0 && buf[len - 1] == '\n') {
    buf[len - 1] = '\0';   // 去掉 \n
    len--;
}
```

```
输入："hello\n"
fgets 读到：h e l l o \n \0    → strlen = 6（包含 \n！）
去掉 \n 后：h e l l o \0       → strlen = 5

输入："hello"  (EOF，没有回车)
fgets 读到：h e l l o \0       → strlen = 5（没有 \n）
```

**为什么不用 `gets`**：`gets` 不管缓冲区大小，输入多长写多长，会导致缓冲区溢出。C11 标准已移除 `gets`。

### 6.10 `printf` 格式控制 — `%.*s` 技巧 ⭐

```c
printf("%.*s", n, str);
```

`%.*s` 中 `*` 表示"宽度从参数中来"，意思是**最多输出 str 的前 n 个字符**。和 `%.ns` 效果一样，但 n 可以是一个变量。

```c
char s[] = "hello";
printf("%.*s", 3, s);    // 输出 "hel"
printf("%.*s", 10, s);   // 输出 "hello"（不足不填充）
printf("%.*s", 0, s);    // 输出 ""（空）

// 对比普通 %.ns（n 必须是字面量）
printf("%.3s", s);       // 输出 "hel"
```

**真题怎么用**：`17-format.c` 用它实现 `%m.ns` 格式的字符串截取。

### 6.11 `isalpha` / `isdigit` / `tolower` / `toupper` — 字符分类与转换

```c
#include <ctype.h>

// === 判断类（返回非0=真，0=假）===
isalpha('a')  → 非0    // 字母 A-Z / a-z
isalpha('_')  → 0      // 下划线不是字母！
isdigit('5')  → 非0    // 数字 0-9
isdigit('x')  → 0
isalnum('x')  → 非0    // 字母或数字
isspace(' ')  → 非0    // 空白字符（空格、\t、\n、\r 等）
islower('a')  → 非0    // 小写字母
isupper('A')  → 非0    // 大写字母

// === 转换类 ===
tolower('A')  → 'a'    // 转小写（非大写字母原样返回）
toupper('a')  → 'A'    // 转大写
```

**真题怎么用**：

```c
// 20-mark.c：判断是否标识符字符
if (isalpha(input[i]) || input[i] == '_') { ... }
else if (isdigit(input[i])) { ... }

// 22-password.c：建立字母存在标记
if (isalpha(input[i])) {
    exist[tolower(input[i]) - 'a'] = 1;  // 统一转小写
}
```

### 6.13 字符串常见错误一览

| 错误 | 示例 | 后果 |
|------|------|------|
| 忘了加 `'\0'` | `s[0]='a';` 后直接用 `strlen` | 读到内存中碰巧的 `\0`，结果随机 |
| `fgets` 没去 `\n` | 读 `"abc\n"` 不处理 | `strlen` = 4，末尾多一个字符 |
| 用 `==` 比较字符串 | `if (s1 == s2)` | 比较的是地址，永远不相等 |
| 忘记 `strcmp` 返回 0 表示相等 | `if (strcmp(a,b))` | 逻辑反了！应该 `== 0` 才相等 |
| `strcpy` 目标太小 | `char s[3]; strcpy(s,"hello");` | 缓冲区溢出，可能崩溃 |
| 数组名不能赋值 | `char s[10]; s = "hello";` | ❌ 编译错误！要用 `strcpy` |
| 混淆 `toupper` 和 `isupper` | 想判断却写成转换 | 逻辑错误 |

### qsort 比较函数模板

```c
// 升序
int cmp(const void *p, const void *q) {
    int x = *(int*)p, y = *(int*)q;
    return (x < y) - (x > y);
}
// 降序：把 return 改成 (x > y) - (x < y)

int a[] = {3, 1, 4, 2};
qsort(a, 4, sizeof(int), cmp);   // → {1, 2, 3, 4}
```

## ❗ 九、高频易错点

1. **`fgets` 保留 `\n`** — 一定要手动去掉！
   ```c
   int len = strlen(buf);
   if (buf[len - 1] == '\n') { buf[len - 1] = '\0'; len--; }
   ```
2. **循环链表用 `do-while`** — 不是 `while`，因为起点就是当前节点
3. **哨兵头节点不在环内** — `head` 是空节点，`head->next` 才是第一个数据
4. **`malloc` 后检查 `NULL`** — 防止内存分配失败
5. **删除节点需要前驱** — 单链表删后继容易，删自己需要前驱
6. **`free` 后不要再访问** — 悬空指针！
7. **边界条件** — 空链表、单节点、最后一个元素
8. **状态机收尾** — 循环结束后要检查 `flag == 1`，把最后一个结果保存
9. **字符串封口** — 手动构建字符串后一定要加 `'\0'`
10. **`strcmp` 返回 0 表示相等** — 不是返回 1！
